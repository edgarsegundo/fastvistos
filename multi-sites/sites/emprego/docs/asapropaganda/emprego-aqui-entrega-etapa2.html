// Slider automático de backgrounds do hero
// Intervalo: 5 segundos · Transição: 800ms (controlada via CSS)
(function () {
  const slides = document.querySelectorAll('.hero__slide-bg');
  const dots   = document.querySelectorAll('.hero__dot');
  let current  = 0;

  function goTo(index) {
    slides[current].classList.remove('active');
    dots[current].classList.remove('active');
    dots[current].setAttribute('aria-selected', 'false');

    current = index;

    slides[current].classList.add('active');
    dots[current].classList.add('active');
    dots[current].setAttribute('aria-selected', 'true');
  }

  // Autoplay
  let timer = setInterval(() => goTo((current + 1) % slides.length), 5000);

  // Clique nos dots
  dots.forEach((dot, i) => {
    dot.addEventListener('click', () => {
      clearInterval(timer);
      goTo(i);
      timer = setInterval(() => goTo((current + 1) % slides.length), 5000);
    });
  });
})();
